    c++指针

    2.1 指针介绍

    • 指针是变量
      • 变量的值是一个地址
    • 指针的值可以存储哪些地址？
      • 另一个变量
      • 地址

    • 举例：int a = 20; 可以声明一个指针指向他

    • 如果指针指向变量或函数，那么为什么不直接用他们？

    • 可以，但不常用：
      • 函数内部，指针可以访问函数作用域外的数据 • 高效操作数组
      • 在堆（heap）上动态分配内存空间
        • 这些空间没有变量名称，只能通过指针访问
    • 可以获取内存中特殊地址
      • 如嵌入式系统、设备驱动等

    声明指针

    // 变量类型 *指针名称
// variable_type *pointer_name;

int *int_ptr; // *靠右
double* double_ptr; // *靠左，编译器都支持
char *char_ptr; // 字符
string *string_ptr; // 字符串

    ******************初始化指针******************

    // 变量类型 *指针名称 {nullptr}; // 初始化为空指针

int *int_ptr {}; // 整数指针初始化为空
double *double_ptr {nullptr}; // 双精度浮点数指针初始化为空
char *char_ptr {nullptr}; // 字符指针初始化为空
string *string_ptr {nullptr}; // 字符串指针初始化为空

    地址操作符&

    #include <iostream>
using namespace std;

int main()
{
    int *ptr;
    cout << "ptr的值是: " << ptr << endl; // 未初始化的指针，值是不确定的
    cout << "ptr的地址是: " << &ptr << endl; // 指针的地址
    cout << "int的大小是: " << sizeof(int) << " bytes" << endl; // int的大小
    cout << "ptr的大小是: " << sizeof(ptr) << " bytes" << endl; // 指针的大小

    ptr = nullptr; // 将ptr设置为空指针
    cout << "ptr的值是: " << ptr << endl; // 0
    cout << "ptr的地址是: " << &ptr << endl; // 指针的地址不变
    return 0;
}

ptr的值是: 0x7fffdfffef40
ptr的地址是: 0x7fffdffdf4f0
int的大小是: 4 bytes
ptr的大小是: 8 bytes
ptr的值是: 0
ptr的地址是: 0x7fffdffdf4f0

    2.2 指针的解引用（dereference）

    获取指针指向的数据—解引用

    #include <iostream>
using namespace std;

int main()
{
    int score {15};
    int *score_ptr {&score};

    cout << "score_ptr指向的值是: " << *score_ptr << endl;

    *score_ptr = 100; // 通过指针修改score的值
    cout << "score_ptr指向的值是: " << *score_ptr << endl;
    cout << "score的值是: " << score << endl; // score的值也被修改了

    return 0;
}

    2.3 动态内存分配

    • 在堆（heap）上程序员自己分配内存空间；
    • 回忆下C++ 数组 （array）
      • 使用数组需要知道数组大小，并且数组大小是固定的；
      • Vector容器大小可以动态调整
    • 使用指针获取堆上刚分配的内存空间

    ******************************************************************************使用 new 关键字 分配内存空间内存空间******************************************************************************

    #include <iostream>
using namespace std;

int main()
{
    int *int_ptr {nullptr};
    cout << "地址是: " << int_ptr << endl; // 输出空指针地址

    int_ptr = new int {10}; // 动态分配内存，初始化为10
    cout << "地址是: " << int_ptr << "，值是: " << *int_ptr << endl; // 输出地址和值

    *int_ptr = 100; // 更改指针指向的值
    cout << "int_ptr的值是: " << *int_ptr << endl; // 输出更改后的值

    return 0;
}
地址是: 0
地址是: 0x55555556b2c0，值是: 10
int_ptr的值是: 100

    使用 delete 关键字 释放内存空间

    #include <iostream>

int main()
{
    int *int_ptr {nullptr}; // 初始化指针为空指针

    int_ptr = new int; // 在堆上动态分配内存

    delete int_ptr; // 释放内存

    return 0;
}

    使用 new[] 为数组分配内存空间

    使用 delete[] 释放内存

    #include <iostream>

int main()
{
    int *array_ptr {nullptr}; // 初始化指针为nullptr
    int arr_size {0}; // 初始化数组大小为0

    cout << "请输入数组大小: ";
    cin >> arr_size;

    array_ptr = new int[arr_size]; // 使用new[]动态分配数组内存

    // 这里省略了其他操作

    delete [] array_ptr; // 使用delete[]释放数组内存

    return 0;
}

    2.4 指针和数组的关系

    • 数组变量名称的值是数组第一个元素的地址；
    • 指针变量的的值是一个地址；
    • 如果指针指向的类型和数组元素的类型一致，那么指针和数组名称几乎是等价的
    #include <iostream>
using namespace std;

int main()
{
    int student_scores[] {100, 95, 90, 86, 84};
    cout << "student_scores的地址是: " << student_scores << endl;
    cout << "第一个元素的地址是: " << &student_scores[0] << endl;
    cout << "*数组方式, 第一个元素的值是: " << student_scores[0] << endl;
    cout << "指针方式, 第一个元素的值是: " << *student_scores << endl;

    int *score_ptr {student_scores};
    cout << "指针的值是: " << score_ptr << endl;
    cout << "指针方式, 第一个元素的值是: " << *score_ptr << endl;

    return 0;
}
student_scores的地址是: 0x7fffffffd2f0
第一个元素的地址是: 0x7fffffffd2f0

指针方式, 第一个元素的值是: 100
指针的值是: 0x7fffffffd2f0

    2.5 const和指针

    • 指针可以操作：地址及被指向的对象
      • 被指向的对象是常量 （pointers to constants）
      • 指针本身是常量 （constant pointers）
      • 指针本身和被指向的对象都是常量 （constant pointers to constants）

    被指向的对象是常量

    • 被指向的对象是常量，不可以通过指针改动
    • 指针本身可以改变
    #include <iostream>

int main()
{
    int high_temp {40};
    int low_temp {20};
    const int *temp_ptr {&high_temp}; // 指向  高温的常量指针

    // *temp_ptr = 100; // 错误，不能通过常量指针修改值
    temp_ptr = &low_temp; // 正确，可以修改指针指向的地址

    return 0;
}

    **************************指针本身是常量**************************

    被指向的对象可以通过指针改动

    指针本身不可以改变

    #include <iostream>

int main()
{
    int high_temp {40};
    int low_temp {20};
    int *const temp_ptr {&high_temp}; // 指向高温的指针常量

    *temp_ptr = 100; // 正确，指针指向的值可以改变
    // temp_ptr = &low_temp; // 错误，不可以改变指针常量的指向

    return 0;
}

    ************************************************************************************************指针本身和被指向的对象都是常量************************************************************************************************

    被指向的对象不可以改动

    指针本身不可以改变

    #include <iostream>

int main()
{
    int high_temp {40};
    int low_temp {20};
    const int *const temp_ptr {&high_temp}; // 定义一个指向整型常量的常量指针

    // temp_ptr = &low_temp; // 错误，不能修改指针常量的指向

    return 0;
}

    2.7 指针函数和指针

    • 函数可以返回指针：
      • Type *function();
    • 返回哪些指针：
      • 函数内部动态分配的内存空间
      • 传入的数据
    • 不能返回函数内的局部变量
    #include <iostream>
using namespace std;

// 返回两个整数中最大的一个
int *max(int *a, int *b)
{
    return (*a > *b) ? a : b;
}

int main()
{
    int x {100}, y {200};
    cout << *max(&x, &y) << endl; // 输出200

    return 0;
}

    #include <cstddef>

// 返回动态分配的内存
int *create_array(size_t size, int initial_value = 0)
{
    int *new_storage{nullptr};

    new_storage = new int[size]; // 分配内存

    for (size_t i{0}; i < size; i++)
    {
        // 初始化值
        *(new_storage + i) = initial_value;
    }

    return new_storage;
}

int main()
{
    // 示例使用，假设函数的其他部分已经实现。
    int *my_array = create_array(10, 42); // 创建包含10个元素，每个都初始化为42的数组
    // ... 使用数组
    delete[] my_array; // 释放分配的内存
    return 0;
}

    不能返回函数内部的局部变量

    // 错误的例子，不要这么做
int *demo_func1(){
    int size {};
    ...
    return &size; // 返回局部变量的地址是错误的，该变量在函数结束后会被销毁
}

// 错误的例子，不要这么做
int *demo_func2(){
    int size {};
    int *int_ptr {&size};
    ...
    return int_ptr; // 返回指向局部变量的指针是错误的，局部变量在函数结束后会被销毁
}

    2.8 指针，引用对比

    • 变量的替身或别名
    • 本质是一个本身是常量的指针
      • Type &name 等同 Type *const name
      • 声明引用时必须初始化（指针变量有地址）
      • 初始化后不可以再作为其它变量的引用（指针本身是常量，不可以修改）
      • 不可以为 null 空
    • 函数经常使用引用传参
    • 引用可以看成一个使用时自动解引用的指针，这个指针他本身是常量
